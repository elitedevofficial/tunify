## Example Usage



```javascript

const { Manager } = require("erela.js");

const filters  = require("erela-custom-filters");





const manager = new Manager({

  plugins: [

    // Initiate the plugin

    new filters()

    

  ]

});

```

### To use filter

```javascript

    // to add nightcore filter

    <Player>.nightcore = true;

    // to remove nightcore filter

    <Player>.nightcore = false;

```

### To reset filter

```javascript

    // to reset filter

    <Player>.reset();

```
